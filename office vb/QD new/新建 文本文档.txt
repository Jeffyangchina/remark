Private Sub waitforms(ms As Long)
Dim s As Long
b = Second(Now)
i = 0
s = b + ms
If s > 59 Then
s = s - 60
End If

Do
b = Second(Now)
ProgressBar1.Value = i
i = i + 0.0008
If i > 29 Then
i = 0
End If
If s = b Then Exit Sub
Loop

End Sub
